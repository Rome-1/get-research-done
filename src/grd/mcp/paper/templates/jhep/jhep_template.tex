%# JHEP template -- Journal of High Energy Physics
%# Jinja2 template with custom LaTeX-safe delimiters
\documentclass[a4paper,11pt]{article}

\usepackage{jheppub}
\usepackage{graphicx}
\usepackage{amsmath}
\usepackage{amssymb}
\usepackage{wasysym}
\providecommand{\tightlist}{\setlength{\itemsep}{0pt}\setlength{\parskip}{0pt}}
% Citation resolution: \citet{key} (textual) and \citep{key} (parenthetical)
% resolve through jheppub, which wraps natbib internally with JHEP-flavoured
% author-year defaults. We do NOT \usepackage{natbib} here -- jheppub already
% pulls it in and a second load triggers "Option clash" errors.
% \bibliographystyle{JHEP} + \bibliography{...} below feed bibtex.

\title{\VAR{title}}

\BLOCK{for author in authors}
\BLOCK{if author.affiliation}
\author[\VAR{loop.index}]{\VAR{author.name}}
\BLOCK{else}
\author{\VAR{author.name}}
\BLOCK{endif}
\BLOCK{endfor}

\BLOCK{for author in authors}
\BLOCK{if author.affiliation}
\affiliation[\VAR{loop.index}]{\VAR{author.affiliation}}
\BLOCK{endif}
\BLOCK{endfor}

\BLOCK{for author in authors}
\BLOCK{if author.email}\emailAdd{\VAR{author.email}}\BLOCK{endif}
\BLOCK{endfor}

\abstract{\VAR{abstract}}

\begin{document}

\maketitle
\flushbottom

\BLOCK{for section in sections}
\section{\VAR{section.title}}
\BLOCK{if section.label}\label{sec:\VAR{section.label}}\BLOCK{endif}
\VAR{section.content}
\BLOCK{endfor}

\BLOCK{for fig in figures}
\BLOCK{if fig.double_column}
\begin{figure*}
\BLOCK{else}
\begin{figure}
\BLOCK{endif}
\centering
\includegraphics[width=\VAR{fig.width}]{\VAR{fig.path}}
\caption{\VAR{fig.caption}}
\label{fig:\VAR{fig.label}}
\BLOCK{if fig.double_column}
\end{figure*}
\BLOCK{else}
\end{figure}
\BLOCK{endif}
\BLOCK{endfor}

\BLOCK{if acknowledgments}
\acknowledgments
\VAR{acknowledgments}
\BLOCK{endif}

\BLOCK{if appendix_sections}
\appendix
\BLOCK{for section in appendix_sections}
\section{\VAR{section.title}}
\BLOCK{if section.label}\label{sec:\VAR{section.label}}\BLOCK{endif}
\VAR{section.content}
\BLOCK{endfor}
\BLOCK{endif}

\bibliographystyle{JHEP}
\bibliography{\VAR{bib_file}}

\BLOCK{if attribution_footer}
\begin{center}
{\footnotesize\textit{\VAR{attribution_footer}}}
\end{center}
\BLOCK{endif}

\end{document}
